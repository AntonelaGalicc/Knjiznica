CREATE TABLE autori (
	id INT PRIMARY KEY AUTO_INCREMENT
    ,ime CHARACTER(10) NOT NULL
    ,prezime CHARACTER(15) NOT NULL
);

CREATE TABLE knjige (
	id INT PRIMARY KEY AUTO_INCREMENT
    ,naslov CHARACTER(10) NOT NULL
    ,godina YEAR NOT NULL
    ,autor_id INT
    ,FOREIGN KEY (autor_id) REFERENCES autori(id)
);
